\subsection*{Ejercicios Propuestos: Dominio}

\begin{exercise}
\problem Encuentre el dominio de $\dfrac{3x+1}{x^2-9}$

\begin{solucion}
Factorizamos: $x^2-9 = (x-3)(x+3)$. El dominio es $\{x|x \neq 3 \text{ y } x \neq -3\}$
\end{solucion}

\problem Encuentre el dominio de $\dfrac{\sqrt{x-3}}{x^2-4x+3}$

\begin{solucion}
El numerador requiere $x \geq 3$. Factorizamos el denominador: $x^2-4x+3 = (x-1)(x-3)$. El dominio es $\{x|x > 3\}$ (excluye $x=3$ por el denominador)
\end{solucion}

\problem Encuentre el dominio de $\dfrac{x^2+1}{x^3-8}$

\begin{solucion}
Factorizamos: $x^3-8 = (x-2)(x^2+2x+4)$. Solo $x=2$ hace cero el denominador. El dominio es $\{x|x \neq 2\}$
\end{solucion}

\problem Encuentre el dominio de $\dfrac{\sqrt{2x+5}}{x^2-7x+10}$

\begin{solucion}
El numerador requiere $2x+5 \geq 0$, es decir $x \geq -\frac{5}{2}$. Factorizamos: $x^2-7x+10 = (x-2)(x-5)$. El dominio es $\{x|x \geq -\frac{5}{2}, x \neq 2, x \neq 5\}$
\end{solucion}

\problem Encuentre el dominio de $\dfrac{x-1}{x^3+x^2-6x}$

\begin{solucion}
Factorizamos: $x^3+x^2-6x = x(x^2+x-6) = x(x+3)(x-2)$. El dominio es $\{x|x \neq 0, x \neq -3, x \neq 2\}$
\end{solucion}
\end{exercise}

\subsection*{Ejercicios Propuestos: Simplificación}

\begin{exercise}
\problem Simplifique $\dfrac{x^2-4}{x^2+5x+6}$

\begin{solucion}
$\dfrac{x^2-4}{x^2+5x+6} = \dfrac{(x-2)(x+2)}{(x+2)(x+3)} = \dfrac{x-2}{x+3}$
\end{solucion}

\problem Simplifique $\dfrac{2x^2-8x}{x^2-16}$

\begin{solucion}
$\dfrac{2x^2-8x}{x^2-16} = \dfrac{2x(x-4)}{(x-4)(x+4)} = \dfrac{2x}{x+4}$
\end{solucion}

\problem Simplifique $\dfrac{x^3-27}{x^2-9}$

\begin{solucion}
$\dfrac{x^3-27}{x^2-9} = \dfrac{(x-3)(x^2+3x+9)}{(x-3)(x+3)} = \dfrac{x^2+3x+9}{x+3}$
\end{solucion}

\problem Simplifique $\dfrac{x^2+6x+9}{x^2-9}$

\begin{solucion}
$\dfrac{x^2+6x+9}{x^2-9} = \dfrac{(x+3)^2}{(x-3)(x+3)} = \dfrac{x+3}{x-3}$
\end{solucion}

\problem Simplifique $\dfrac{3x^2-12}{x^2-4x+4}$

\begin{solucion}
$\dfrac{3x^2-12}{x^2-4x+4} = \dfrac{3(x^2-4)}{(x-2)^2} = \dfrac{3(x-2)(x+2)}{(x-2)^2} = \dfrac{3(x+2)}{x-2}$
\end{solucion}
\end{exercise}

\subsection*{Ejercicios Propuestos: Multiplicación}

\begin{exercise}
\problem Multiplique $\dfrac{x^2-1}{x^2+3x+2} \cdot \dfrac{x+2}{x-1}$

\begin{solucion}
$\dfrac{(x-1)(x+1)}{(x+1)(x+2)} \cdot \dfrac{x+2}{x-1} = \dfrac{\cancel{(x-1)}\cancel{(x+1)}}{\cancel{(x+1)}\cancel{(x+2)}} \cdot \dfrac{\cancel{x+2}}{\cancel{x-1}} = 1$
\end{solucion}

\problem Multiplique $\dfrac{2x^2-2x-12}{x^2-9} \cdot \dfrac{x+3}{4x-12}$

\begin{solucion}
$\dfrac{2(x^2-x-6)}{(x-3)(x+3)} \cdot \dfrac{x+3}{4(x-3)} = \dfrac{2(x-3)(x+2)}{(x-3)(x+3)} \cdot \dfrac{x+3}{4(x-3)} = \dfrac{x+2}{2(x-3)}$
\end{solucion}

\problem Multiplique $\dfrac{x^2-4}{x+3} \cdot \dfrac{x^2+6x+9}{x^2-4x+4}$

\begin{solucion}
$\dfrac{(x-2)(x+2)}{x+3} \cdot \dfrac{(x+3)^2}{(x-2)^2} = \dfrac{\cancel{(x-2)}(x+2)\cancel{(x+3)}^2}{\cancel{x+3}(x-2)^{\cancel{2}}} = \dfrac{(x+2)(x+3)}{x-2}$
\end{solucion}

\problem Multiplique $\dfrac{x^2+x-6}{x^2-1} \cdot \dfrac{x^2-2x+1}{x^2-9}$

\begin{solucion}
$\dfrac{(x+3)(x-2)}{(x-1)(x+1)} \cdot \dfrac{(x-1)^2}{(x-3)(x+3)} = \dfrac{\cancel{(x+3)}(x-2)\cancel{(x-1)}}{\cancel{(x-1)}(x+1)(x-3)\cancel{(x+3)}} = \dfrac{x-2}{(x+1)(x-3)}$
\end{solucion}

\problem Multiplique $\dfrac{2x^2-8}{x^2+5x} \cdot \dfrac{x^2+4x-5}{x-2}$

\begin{solucion}
$\dfrac{2(x^2-4)}{x(x+5)} \cdot \dfrac{(x+5)(x-1)}{x-2} = \dfrac{2(x-2)(x+2)\cancel{(x+5)}(x-1)}{x\cancel{(x+5)}\cancel{(x-2)}} = \dfrac{2(x+2)(x-1)}{x}$
\end{solucion}
\end{exercise}

\subsection*{Ejercicios Propuestos: División}

\begin{exercise}
\problem Divida $\dfrac{x^2-9}{x^2+4x+4} \div \dfrac{x+3}{x+2}$

\begin{solucion}
$\dfrac{x^2-9}{x^2+4x+4} \cdot \dfrac{x+2}{x+3} = \dfrac{(x-3)(x+3)}{(x+2)^2} \cdot \dfrac{x+2}{x+3} = \dfrac{(x-3)\cancel{(x+3)}\cancel{(x+2)}}{(x+2)^{\cancel{2}}\cancel{(x+3)}} = \dfrac{x-3}{x+2}$
\end{solucion}

\problem Divida $\dfrac{x^2+5x+6}{x^2-4} \div \dfrac{x^2+4x+3}{x^2-3x+2}$

\begin{solucion}
$\dfrac{(x+2)(x+3)}{(x-2)(x+2)} \cdot \dfrac{(x-1)(x-2)}{(x+1)(x+3)} = \dfrac{\cancel{(x+2)}\cancel{(x+3)}\cancel{(x-2)}(x-1)}{\cancel{(x-2)}\cancel{(x+2)}(x+1)\cancel{(x+3)}} = \dfrac{x-1}{x+1}$
\end{solucion}

\problem Divida $\dfrac{x^2-1}{x+3} \div \dfrac{x-1}{x^2+6x+9}$

\begin{solucion}
$\dfrac{x^2-1}{x+3} \cdot \dfrac{x^2+6x+9}{x-1} = \dfrac{(x-1)(x+1)}{x+3} \cdot \dfrac{(x+3)^2}{x-1} = \dfrac{\cancel{(x-1)}(x+1)\cancel{(x+3)}^2}{\cancel{x+3}\cancel{(x-1)}} = (x+1)(x+3)$
\end{solucion}

\problem Divida $\dfrac{2x^2-8}{x^2+x-6} \div \dfrac{x-2}{x+3}$

\begin{solucion}
$\dfrac{2x^2-8}{x^2+x-6} \cdot \dfrac{x+3}{x-2} = \dfrac{2(x-2)(x+2)}{(x+3)(x-2)} \cdot \dfrac{x+3}{x-2} = \dfrac{2\cancel{(x-2)}(x+2)\cancel{(x+3)}}{\cancel{(x+3)}\cancel{(x-2)}^2} = \dfrac{2(x+2)}{x-2}$
\end{solucion}

\problem Simplifique $\dfrac{x^2-16}{2x+6} \cdot \dfrac{x^2+3x}{x^2-4x} \div \dfrac{x+4}{2x}$

\begin{solucion}
$\dfrac{(x-4)(x+4)}{2(x+3)} \cdot \dfrac{x(x+3)}{x(x-4)} \cdot \dfrac{2x}{x+4} = \dfrac{\cancel{(x-4)}\cancel{(x+4)}\cancel{x}\cancel{(x+3)}2\cancel{x}}{2\cancel{(x+3)}\cancel{x}\cancel{(x-4)}\cancel{(x+4)}} = 1$
\end{solucion}
\end{exercise}

\subsection*{Ejercicios Propuestos: Suma y Resta}

\begin{exercise}
\problem Efectúe $\dfrac{3}{x-2} + \dfrac{4}{x+3}$

\begin{solucion}
$MCD = (x-2)(x+3)$
$$\frac{3(x+3)}{(x-2)(x+3)} + \frac{4(x-2)}{(x-2)(x+3)} = \frac{3x+9+4x-8}{(x-2)(x+3)} = \frac{7x+1}{(x-2)(x+3)}$$
\end{solucion}

\problem Efectúe $\dfrac{x}{x^2-4} - \dfrac{2}{x+2}$

\begin{solucion}
$MCD = (x-2)(x+2)$
$$\frac{x}{(x-2)(x+2)} - \frac{2(x-2)}{(x-2)(x+2)} = \frac{x-2x+4}{(x-2)(x+2)} = \frac{-x+4}{(x-2)(x+2)}$$
\end{solucion}

\problem Efectúe $\dfrac{1}{x} + \dfrac{2}{x^2} - \dfrac{3}{x^3}$

\begin{solucion}
$MCD = x^3$
$$\frac{x^2}{x^3} + \frac{2x}{x^3} - \frac{3}{x^3} = \frac{x^2+2x-3}{x^3} = \frac{(x+3)(x-1)}{x^3}$$
\end{solucion}

\problem Efectúe $\dfrac{2x}{x^2-1} + \dfrac{3}{x-1} - \dfrac{1}{x+1}$

\begin{solucion}
$MCD = (x-1)(x+1)$
\begin{align*}
&= \frac{2x}{(x-1)(x+1)} + \frac{3(x+1)}{(x-1)(x+1)} - \frac{(x-1)}{(x-1)(x+1)}\\
&= \frac{2x+3x+3-x+1}{(x-1)(x+1)} = \frac{4x+4}{(x-1)(x+1)} = \frac{4(x+1)}{(x-1)(x+1)} = \frac{4}{x-1}
\end{align*}
\end{solucion}

\problem Efectúe $\dfrac{x+1}{x^2+5x+6} - \dfrac{x-1}{x^2+4x+3}$

\begin{solucion}
Factorizamos: $x^2+5x+6 = (x+2)(x+3)$ y $x^2+4x+3 = (x+1)(x+3)$

$MCD = (x+1)(x+2)(x+3)$
\begin{align*}
&= \frac{(x+1)(x+1)}{(x+1)(x+2)(x+3)} - \frac{(x-1)(x+2)}{(x+1)(x+2)(x+3)}\\
&= \frac{x^2+2x+1-(x^2+x-2)}{(x+1)(x+2)(x+3)}\\
&= \frac{x+3}{(x+1)(x+2)(x+3)} = \frac{1}{(x+1)(x+2)}
\end{align*}
\end{solucion}
\end{exercise}

\subsection*{Ejercicios Propuestos: Fracciones Compuestas}

\begin{exercise}
\problem Simplifique $\dfrac{1+\dfrac{1}{x}}{1-\dfrac{1}{x^2}}$

\begin{solucion}
$$\frac{\dfrac{x+1}{x}}{\dfrac{x^2-1}{x^2}} = \frac{x+1}{x} \cdot \frac{x^2}{x^2-1} = \frac{(x+1)x^2}{x(x-1)(x+1)} = \frac{x}{x-1}$$
\end{solucion}

\problem Simplifique $\dfrac{\dfrac{a}{b}+\dfrac{b}{a}}{\dfrac{1}{a}+\dfrac{1}{b}}$

\begin{solucion}
$$\frac{\dfrac{a^2+b^2}{ab}}{\dfrac{a+b}{ab}} = \frac{a^2+b^2}{ab} \cdot \frac{ab}{a+b} = \frac{a^2+b^2}{a+b}$$
\end{solucion}

\problem Simplifique $\dfrac{\dfrac{x+1}{x-1}}{\dfrac{x^2-1}{x+2}}$

\begin{solucion}
$$\frac{x+1}{x-1} \div \frac{x^2-1}{x+2} = \frac{x+1}{x-1} \cdot \frac{x+2}{(x-1)(x+1)} = \frac{\cancel{(x+1)}(x+2)}{(x-1)^2\cancel{(x+1)}} = \frac{x+2}{(x-1)^2}$$
\end{solucion}

\problem Simplifique $\dfrac{1-\dfrac{2}{x}+\dfrac{1}{x^2}}{1-\dfrac{1}{x^2}}$

\begin{solucion}
$$\frac{\dfrac{x^2-2x+1}{x^2}}{\dfrac{x^2-1}{x^2}} = \frac{x^2-2x+1}{x^2} \cdot \frac{x^2}{x^2-1} = \frac{(x-1)^2}{(x-1)(x+1)} = \frac{x-1}{x+1}$$
\end{solucion}

\problem Simplifique $\dfrac{\dfrac{1}{x+1}-\dfrac{1}{x-1}}{\dfrac{2}{x^2-1}}$

\begin{solucion}
$$\frac{\dfrac{(x-1)-(x+1)}{(x+1)(x-1)}}{\dfrac{2}{x^2-1}} = \frac{\dfrac{-2}{x^2-1}}{\dfrac{2}{x^2-1}} = \frac{-2}{x^2-1} \cdot \frac{x^2-1}{2} = -1$$
\end{solucion}
\end{exercise}

\subsection*{Ejercicios Adicionales de Repaso}

\begin{exercise}
\problem Simplifique completamente: $\dfrac{x^3-8}{x^2-4} \cdot \dfrac{x+2}{x^2+2x+4}$

\begin{solucion}
$$\frac{(x-2)(x^2+2x+4)}{(x-2)(x+2)} \cdot \frac{x+2}{x^2+2x+4} = \frac{\cancel{(x-2)}\cancel{(x^2+2x+4)}\cancel{(x+2)}}{\cancel{(x-2)}\cancel{(x+2)}\cancel{(x^2+2x+4)}} = 1$$
\end{solucion}

\problem Efectúe y simplifique: $\dfrac{1}{x+1} + \dfrac{2}{x-1} - \dfrac{3x}{x^2-1}$

\begin{solucion}
$MCD = (x-1)(x+1)$
\begin{align*}
&= \frac{(x-1)}{(x+1)(x-1)} + \frac{2(x+1)}{(x+1)(x-1)} - \frac{3x}{(x+1)(x-1)}\\
&= \frac{x-1+2x+2-3x}{(x+1)(x-1)} = \frac{1}{(x+1)(x-1)}
\end{align*}
\end{solucion}

\problem Simplifique la fracción compuesta: $\dfrac{\dfrac{1}{a}-\dfrac{1}{b}}{\dfrac{1}{a^2}-\dfrac{1}{b^2}}$

\begin{solucion}
$$\frac{\dfrac{b-a}{ab}}{\dfrac{b^2-a^2}{a^2b^2}} = \frac{b-a}{ab} \cdot \frac{a^2b^2}{(b-a)(b+a)} = \frac{\cancel{(b-a)}ab\cancel{ab}}{\cancel{ab}\cancel{(b-a)}(b+a)} = \frac{ab}{b+a}$$
\end{solucion}

\problem Encuentre el dominio y simplifique: $\dfrac{x^2-5x+6}{x^2-9}$

\begin{solucion}
Dominio: $x^2-9 = (x-3)(x+3)$, entonces $\{x|x \neq 3, x \neq -3\}$

Simplificación: $\dfrac{(x-2)(x-3)}{(x-3)(x+3)} = \dfrac{x-2}{x+3}$ (con $x \neq 3, x \neq -3$)
\end{solucion}
\end{exercise}
